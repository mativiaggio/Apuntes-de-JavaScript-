Switch

let food = prompt("Que se te antoja cenar?: ")
switch (food) {
    case 'hamburguesa':
        console.log(`Pedido: ${food}, valor $1200.`)
    break;
    case 'pizza':
        console.log(`Pedido: ${food}, valor $900.`)
    break
    case 'lomo':
        console.log(`Pedido: ${food}, valor $800.`)
    break
    default:
        console.log('Te ahorras unos pesos si no pedis.')
    break
}
